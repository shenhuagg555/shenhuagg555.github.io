BinaryTree

Data structure and algorithm
发布日期:   2020-06-15

二叉节点的创建

struct BinNode{
    int val;
    struct BinNode *left, *right;
    BinNode(int x){              //直接一个有参构造方法方便节点创建
        val = x;
        left = right = NULL;
    }
};

前序遍历

void preOrder(BinNode *root){       //前序遍历
    if(root){
        cout << root->val << endl;
        preOrder(root->left);
        preOrder(root->right);
    }
}

中序遍历

void inOrder(BinNode *root){        //中序遍历
    if(root){
        inOrder(root->left);
        cout << root->val << endl;
        inOrder(root->right);
    }
}

后序遍历

void postOrder(BinNode *root){       //后序遍历
    if(root){
        postOrder(root->left);
        postOrder(root->right);
        cout << root->val << endl;
    }
}

二叉树的深度

int getDepth(BinNode *root){          //获取树的深度
    if(root){
        int leftdepth = getDepth(root->left);
        int rightdepth = getDepth(root->right);
        return leftdepth > rightdepth ? leftdepth + 1 : rightdepth + 1;   //比较左右子树的深度再加上此节点
    }
    else
        return 0;            //若为空节点则返回0，相当于没有增加树的深度
}

统计叶子节点数

int leafCount(BinNode *root){            //统计叶子节点
    if(root){
        if(root->left == NULL && root->right == NULL)        //若节点的左右子树均为空，则返回1(即自身为叶子节点)
            return 1;
        else
            return leafCount(root->left) + leafCount(root->right);   //若节点左右子树不全为空，则返回左右子树的叶子节点书之和
    }
    else
        return 0;        //返回0表示对叶子节点个数贡献为0
}

统计节点数

int nodeCount(BinNode *root){           //统计节点个书
    if(root)
        return nodeCount(root->left) + nodeCount(root->right) + 1;        //返回左右子树节点数之和再加此节点
    else
        return 0;   //返回0表示对节点个数贡献为0
}

搜索节点

BinNode *searchNode(BinNode *root, int val){          //查找节点
    if(root == NULL)        //空节点直接返回NULL，（或代表节点未找到）
        return NULL;
    if(root->val == val){   //找到目标节点，返回其地址
        return root;
    }
    BinNode *p = searchNode(root->left, val);        //前面的程序没有找到，则查找该节点的左子节点，若为空则继续查找右子节点
    return p ? p : searchNode(root->right, val);
}

插入节点

bool insertNode(BinNode *root, int val, char LR, int x){   //插入节点(不牵扯递归，都可以看懂)
    BinNode *p = searchNode(root, val);
    if(p == NULL)
        return false;
    if(LR == 'L'){
        if(p->left != NULL)
            return false;
        p->left = new BinNode(x);
    }
    else{
        if(p->right != NULL)
            return false;
        p->right = new BinNode(x);
    }
    return true;
}

二叉树的删除

void clearTree(BinNode* &root){           //整树删除
    if(root){                      //若节点不为空，则依次删除其左右子树
        clearTree(root->left);
        clearTree(root->right);
        delete root;            //调用delete 删除节点
        root = NULL;           //记得将被删除节点的指针指向NULL，否则将变为“野指针”
    }
}
